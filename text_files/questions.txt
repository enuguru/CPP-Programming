convert the excel column ids to decimal


#include <iostream>
using namespace std;
#include <bits/stdc++.h> 
#include<string>
#include<math.h>
int main()
{
    string n="CBD";
    int ans=0,j=0;
    int l=n.length();
    for(int i=l;i>0;i--)
    {
        ans=ans+ (n[i]-"A"*pow(26,j));
        j++;
    }
    cout << "decimal vale for CBD-- "<<ans;
    return 0;
}
